        analyzeSignal(rssi: number): string {
            let signalHtml = 'disconnected';

            if (rssi >= -65) {
            signalHtml = 'full';
            } else if (rssi >= -75 && rssi < -65) {
            signalHtml = 'good';
            } else if (rssi >= -85 && rssi < -75) {
            signalHtml = 'fair';
            } else if (rssi >= -95) {
            signalHtml = 'poor';
            }

            return signalHtml;
        }
