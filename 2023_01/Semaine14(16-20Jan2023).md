                ENV KC_DB_USERNAME=<DBUSERNAME>
                ENV KC_DB_PASSWORD=<DBPASSWORD>
                ENV KC_HOSTNAME=localhost
                ENTRYPOINT ["/opt/keycloak/bin/kc.sh"]
